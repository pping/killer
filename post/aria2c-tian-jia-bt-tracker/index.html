aria2c 添加bt-tracker

2021-02-20

方法1

vim aria.sh

#!/bin/bash
tracker=`curl -s -k https://ngosang.github.io/trackerslist/trackers_best.txt  |awk NF|sed ":a;N;s/\n/,/g;b a"`
#echo $tracker
aria2c $* --bt-tracker=$tracker

chmod +x aria.sh

./aria.sh
